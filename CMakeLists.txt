PROJECT (VMC_OPENMP_VER)
CMAKE_MINIMUM_REQUIRED (VERSION 3.0 FATAL_ERROR)

SET (TARGET train_nqs)
SET (SRC ${TARGET}.cpp)
IF (APPLE)
  SET (CMAKE_CXX_COMPILER /usr/local/opt/llvm/bin/clang++)
ELSEIF (UNIX)
  IF (NOT CMAKE_CXX_COMPILER)
    SET (CMAKE_CXX_COMPILER /usr/bin/g++)
  ENDIF (NOT CMAKE_CXX_COMPILER)
ENDIF ()
SET (CMAKE_CXX_FLAGS "-std=c++11 -O3 -fopenmp")
INCLUDE (${CMAKE_CURRENT_SOURCE_DIR}/cmake/FindMKL.cmake)
INCLUDE (${CMAKE_CURRENT_SOURCE_DIR}/cmake/FindTRNG4.cmake)
LINK_LIBRARIES (${MKL_LIBRARIES} ${MKL_OPENMP_LIBRARY} ${TRNG4_LIBRARIES})
ADD_EXECUTABLE (${TARGET} ${SRC})
