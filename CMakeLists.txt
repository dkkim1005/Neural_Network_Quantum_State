IF (APPLE)
  # Plz check that clang++ is installed at the below directory.
  SET (CMAKE_CXX_COMPILER /usr/local/opt/llvm/bin/clang++)
ELSEIF (UNIX)
  # You can specify the compiler you want via 'CMAKE_CXX_COMPILER' variable.
  IF (NOT CMAKE_CXX_COMPILER)
    SET (CMAKE_CXX_COMPILER /usr/bin/g++)
  ENDIF (NOT CMAKE_CXX_COMPILER)
ENDIF ()

#SET (USE_CUDA TRUE) # --> CUDA 10.x version is required(experimental).
PROJECT (VMC_OPENMP_VER)
CMAKE_MINIMUM_REQUIRED (VERSION 3.0 FATAL_ERROR)
SET (TARGETS "SQ-train_fnn;CB-train_fnn")
INCLUDE (${CMAKE_CURRENT_SOURCE_DIR}/cmake/FindMKL.cmake)
INCLUDE (${CMAKE_CURRENT_SOURCE_DIR}/cmake/FindTRNG4.cmake)
IF (USE_CUDA)
  FIND_PACKAGE (CUDA)
  INCLUDE (${CMAKE_CURRENT_SOURCE_DIR}/cmake/FindMAGMA.cmake)
  SET (CUDA_NVCC_FLAGS "-O3 -Xcompiler -fopenmp -std=c++11 -DMIN_CUDA_ARCH=700")
  SET (CUDA_LIBRARIES ${CUDA_LIBRARIES} "cublas" "cusparse" "cudart" "cudadevrt")
  SET (CUDA_LIBRARIES ${CUDA_LIBRARIES} ${MAGMA_LIBRARIES})
ELSE ()
  SET (CMAKE_CXX_FLAGS "-std=c++14 -O3 -fopenmp")
ENDIF ()
LINK_LIBRARIES (${MKL_LIBRARIES} ${MKL_OPENMP_LIBRARY} ${TRNG4_LIBRARIES} ${CUDA_LIBRARIES})
FOREACH (TARGET ${TARGETS})
  IF (USE_CUDA)
    CUDA_ADD_EXECUTABLE (${TARGET}-gpu ${CMAKE_CURRENT_SOURCE_DIR}/gpu/${TARGET}.cu)
  ELSE ()
    ADD_EXECUTABLE (${TARGET}-cpu ${CMAKE_CURRENT_SOURCE_DIR}/cpu/${TARGET}.cpp)
  ENDIF ()
ENDFOREACH ()
